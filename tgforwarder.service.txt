[Unit]
Description=Telegram AutoForwarder
After=network.target

[Service]
Type=simple
User=pi
WorkingDirectory=/home/pi/Telegram-Autoforwarder
# run under bash -lc so that "source" and "&&" chaining works
ExecStart=/bin/bash -lc '\
  cd /home/pi/Telegram-Autoforwarder && \
  . .venv/bin/activate && \
  python3 TelegramForwarder_autorun.py \
'
Restart=on-failure
RestartSec=5
StandardOutput=journal
StandardError=journal

[Install]
WantedBy=multi-user.target
